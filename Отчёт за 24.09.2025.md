## –ü—Ä–∏–ª–æ–∂–µ–Ω–∏–µ: –∫–æ–¥ –ø—Ä–æ–≥—Ä–∞–º–º—ã

```python
import tkinter as tk
from tkinter import messagebox
import sqlite3


# --- –†–∞–±–æ—Ç–∞ —Å –±–∞–∑–æ–π –¥–∞–Ω–Ω—ã—Ö ---
def init_db():
    conn = sqlite3.connect("tasks.db")
    cursor = conn.cursor()
    cursor.execute("""
        CREATE TABLE IF NOT EXISTS tasks (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            title TEXT NOT NULL,
            description TEXT,
            date TEXT,
            done INTEGER DEFAULT 0
        )
    """)
    conn.commit()
    conn.close()


def add_task_db(title, description, date):
    conn = sqlite3.connect("tasks.db")
    cursor = conn.cursor()
    cursor.execute("INSERT INTO tasks (title, description, date, done) VALUES (?, ?, ?, ?)",
                   (title, description, date, 0))
    conn.commit()
    conn.close()


def get_tasks():
    """–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç —Å–ø–∏—Å–æ–∫ –∫–æ—Ä—Ç–µ–∂–µ–π (id, title, description, date, done) ‚Äî —É–ø–æ—Ä—è–¥–æ—á–µ–Ω–Ω—ã—Ö –ø–æ id."""
    conn = sqlite3.connect("tasks.db")
    cursor = conn.cursor()
    cursor.execute("SELECT * FROM tasks ORDER BY id ASC")
    tasks = cursor.fetchall()
    conn.close()
    return tasks


def mark_done(task_id):
    conn = sqlite3.connect("tasks.db")
    cursor = conn.cursor()
    cursor.execute("UPDATE tasks SET done = 1 WHERE id = ?", (task_id,))
    conn.commit()
    conn.close()


def delete_task_db(task_id):
    conn = sqlite3.connect("tasks.db")
    cursor = conn.cursor()
    cursor.execute("DELETE FROM tasks WHERE id = ?", (task_id,))
    conn.commit()
    conn.close()


# --- –ò–Ω—Ç–µ—Ä—Ñ–µ–π—Å Tkinter ---
class TaskApp:
    def __init__(self, root):

        self.title_entry = tk.Entry(frame_add, width=40, relief="solid", bd=1, font=("Arial", 12))

        tk.Label(frame_add, text="–û–ø–∏—Å–∞–Ω–∏–µ:", bg="#f2f2f2").pack(anchor="w")
        self.desc_entry = tk.Entry(frame_add, width=40, relief="solid", bd=1, font=("Arial", 12))

        frame_list = tk.Frame(root, bg="#f2f2f2", padx=10, pady=10)

        tk.Label(frame_list, text="–°–ø–∏—Å–æ–∫ –∑–∞–¥–∞—á", font=("Arial", 16, "bold"), bg="#f2f2f2").pack(pady=5)

    def add_task(self):

        desc = self.desc_entry.get()
        date = self.date_entry.get()

            messagebox.showwarning("–û—à–∏–±–∫–∞", "–ù–∞–∑–≤–∞–Ω–∏–µ –∑–∞–¥–∞—á–∏ –Ω–µ –º–æ–∂–µ—Ç –±—ã—Ç—å –ø—É—Å—Ç—ã–º")

                                      f"–î–æ–±–∞–≤–∏—Ç—å –∑–∞–¥–∞—á—É:\n\n–ù–∞–∑–≤–∞–Ω–∏–µ: {title}\n–û–ø–∏—Å–∞–Ω–∏–µ: {desc}\n–î–∞—Ç–∞: {date}?")

            self.desc_entry.delete(0, tk.END)
            self.date_entry.delete(0, tk.END)

    def load_tasks(self):
        """–ó–∞–≥—Ä—É–∂–∞–µ–º –∑–∞–¥–∞—á–∏ –∏–∑ –ë–î –≤ self.tasks –∏ –æ—Ç–æ–±—Ä–∞–∂–∞–µ–º –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω—É—é –Ω—É–º–µ—Ä–∞—Ü–∏—é 1..N"""
        self.task_listbox.delete(0, tk.END)
        self.tasks = get_tasks()  # —Å–ø–∏—Å–æ–∫ –∫–æ—Ä—Ç–µ–∂–µ–π
        for idx, task in enumerate(self.tasks, start=1):
            status = "‚úì" if task[4] == 1 else "‚úó"
            # –û—Ç–æ–±—Ä–∞–∂–∞–µ–º –Ω–æ–º–µ—Ä –∫–∞–∫ idx (–ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω—ã–π), –∞ –Ω–µ –∫–∞–∫ task[0] (id –≤ –ë–î)
            self.task_listbox.insert(tk.END, f"{idx}. {task[1]} ({task[3]}) [{status}]")

    def done_task(self):
        """–ë–µ—Ä—ë–º –∏–Ω–¥–µ–∫—Å –≤—ã–±—Ä–∞–Ω–Ω–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞, –Ω–∞—Ö–æ–¥–∏–º real id –≤ self.tasks –∏ –ø–æ–º–µ—á–∞–µ–º –≤—ã–ø–æ–ª–Ω–µ–Ω–Ω—ã–º"""
        sel = self.task_listbox.curselection()
        if not sel:
            messagebox.showwarning("–û—à–∏–±–∫–∞", "–í—ã–±–µ—Ä–∏—Ç–µ –∑–∞–¥–∞—á—É")
            return
        index = sel[0]
        task = self.tasks[index]  # –∫–æ—Ä—Ç–µ–∂ (id, title, desc, date, done)
        task_id, title, *_ = task

        confirm = messagebox.askyesno("–ü–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ", f"–û—Ç–º–µ—Ç–∏—Ç—å –∑–∞–¥–∞—á—É '{title}' –∫–∞–∫ –≤—ã–ø–æ–ª–Ω–µ–Ω–Ω—É—é?")
        if confirm:
            mark_done(task_id)
            self.load_tasks()

    def delete_task(self):
        sel = self.task_listbox.curselection()
        if not sel:
            messagebox.showwarning("–û—à–∏–±–∫–∞", "–í—ã–±–µ—Ä–∏—Ç–µ –∑–∞–¥–∞—á—É")
            return
        index = sel[0]
        task = self.tasks[index]
        task_id, title, *_ = task

        confirm = messagebox.askyesno("–ü–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–∏–µ", f"–£–¥–∞–ª–∏—Ç—å –∑–∞–¥–∞—á—É '{title}'?")
        if confirm:
            delete_task_db(task_id)
            self.load_tasks()
```
